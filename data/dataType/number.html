The Number Type

  • Internally a number is represented as a 64-bit floating-point. Javascript does not make a difference between integer and a floating point.


var foo = "";         //foo is defined.
foo = 10;        //foo is now a number

var bar = 10;     //foo

            

var foo;         //foo is defined, but its type is still undefined
foo = 10;        //foo is now a number
foo = "abc";     //foo is now a string
foo = true;      //foo is now a boolean

            

// the boolean type
foo = true;        //foo is a boolean with true
foo = false;     //foo is a boolean with a false


foo = "abc";     //foo is now a string
foo = true;      //foo is now a boolean
